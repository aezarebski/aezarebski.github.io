Objects and Classes

Notes

Here is a little example of how to define a class and how inheritance works in JavaScript 

class Shape {
    constructor (name, area) {
	this.name = name;
	this.area = area;
    }

    greet () {
	console.log(`Hello, my name is ${this.name}.`);
	console.log(`My area is ${this.area}`);
    }
}

class Square extends Shape {
    constructor (sideLength) {
	super('square', sideLength * sideLength);
	this.sideLength = sideLength;
    }

    elaborate () {
	console.log(`My side length is ${this.sideLength}`);
    }
}

const sq = new Square(3);
sq.greet();
sq.elaborate();

Exercises

Delays

// delays.js
class Delay {
    constructor (x) {
	this.nextX = x;
	this.currX = null;
    }

    call (newX) {
	this.currX = this.nextX;
	this.nextX = newX;
	return this.currX;
    }
}

const example = new Delay('a');
for (let value of ['b', 'c', 'd']) {
    console.log(value, '->', example.call(value));
}

$ node delays.js
b -> a
c -> b
d -> c

Filtering

// filtering.js
class Filter {
    constructor (...values) {
	this.values = values;
    }
    call (x) {
	return this.values.some((v) => x === v) ? null : x;
    }
}

const example = new Filter('a', 'e', 'i', 'o', 'u');
for (let value of ['a', 'b', 'c', 'd', 'e']) {
    console.log(value, '->', example.call(value));
}

$ node filetering.js
a -> null
b -> b
c -> c
d -> d
e -> null

Pipelines

My first attempt at this reduced the array of callable objects but the requirement to short circuit upon a null result complicated things. The loop seems more readable and simpler. 

// pipeline.js

// class Delay { } as before

// class Filter {} as before

class Pipeline {
    constructor (...callables) {
	this.callables = callables;
    }

    call (x) {
	var y = x;
	for (let obj of this.callables) {
	    y = obj.call(y);
	    if (y == null) {
		return null;
	    }
	}
	return y;
    }
}

const example = new Pipeline(new Filter('a', 'e', 'i', 'o', 'u'),
			     new Delay('a'));
for (let value of ['a' ,'b', 'c', 'd', 'e']) {
    console.log(value, '->', example.call(value));
}

$ node pipeline.js
a -> null
b -> a
c -> b
d -> c
e -> null

Active Expressions

The last part of this question looking at the delayed data just requires using the result above and if you want to get fancy, you can have it inherit from Active and overwrite some of the methods. 

// active.js

class Active {

    constructor (name, transFunc = null) {
	this.name = name;
	this.transFunc = transFunc;
	this.subscribers = [];
    }

    subscribe (someone) {
	this.subscribers.push(someone);
    }

    transform (input) {
	return Math.sqrt(input);
    }

    update (input) {
	console.log(this.name, 'got', input);
	const output = this.transFunc == null ? this.transform(input) : this.transFunc(input);
	for (let s of this.subscribers) {
	    s.update(output);
	}
    }
}

const start = new Active('start', (x) => Math.min(x, 10));
// const left = new Active('left', (x) => 2 * x);
const left = new Active('left');
const right = new Active('right', (x) => x + 1);
const final = new Active('final', (x) => x);
start.subscribe(left);
start.subscribe(right);
left.subscribe(final);
right.subscribe(final);

start.update(123);

$ node active.js
start got 123
left got 10
final got 3.1622776601683795
right got 10
final got 11
