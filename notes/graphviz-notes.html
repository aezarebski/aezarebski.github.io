Example: Compartmental diagram

// dot :tangle ../misc/graphvis-sir-example.gv
// graphviz-sir-example.gv

digraph G {

    rankdir=LR;

    Susceptible [shape=box, color=green];
    Infectious [style=filled];

    Susceptible -> Infectious [style=dotted, label="Infection"];
    Infectious -> Recovered [color=red];

}

dot -Tpng graphvis-sir-example.gv -o graphviz-sir-example.png

Example: Hyperlinks

Click the nodes to go to their Wikipedia page 

// dot :tangle ../misc/graphvis-hyperlink-example.gv
// graphvis-hyperlink-example.gv

digraph G {

    rankdir=LR;

    GraphViz [shape=box, URL="https://en.wikipedia.org/wiki/Graphviz"];
    DOT [shape=box, URL="https://en.wikipedia.org/wiki/DOT_(graph_description_language)"];

    DOT -> GraphViz;

}

dot -Tsvg graphvis-hyperlink-example.gv -o ../resources/graphvis-hyperlink-example.svg

Example: Fancy diagram

// dot :tangle ../misc/graphvis-hyperlink-example.gv
// graphvis-fancy-example.gv

digraph G {
    rankdir=LR;

    node [style="rounded,filled",
          shape=box,
          fillcolor="#c994c7",
          color="#756bb1",
          penwidth=3];

    "epi-types" [URL="https://github.com/aezarebski/epi-types"];
    "epi-sim" [URL="https://github.com/aezarebski/epi-sim"];
    "bdscod" [URL="https://github.com/aezarebski/bdscod"];
    "bdscod-mcmc" [URL="https://github.com/aezarebski/bdscod-mcmc"];
    "mighty-metropolis" [URL="https://github.com/aezarebski/mighty-metropolis"];

    "epi-types" -> "epi-sim" [arrowhead=open, color="#756bb1", penwidth=3];
    "epi-types" -> "bdscod" [arrowhead=open, color="#756bb1", penwidth=3];
    "epi-sim" -> "bdscod" [arrowhead=open, color="#756bb1", penwidth=3];
    "epi-sim" -> "bdscod-mcmc" [arrowhead=open, color="#756bb1", penwidth=3];
    "bdscod" -> "bdscod-mcmc" [arrowhead=open, color="#756bb1", penwidth=3];
    "mighty-metropolis" -> "bdscod-mcmc" [arrowhead=open, color="#756bb1", penwidth=3];
}

dot -Tsvg graphvis-fancy-example.gv -o ../resources/graphvis-fancy-example.svg
